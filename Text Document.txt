```python
import numpy as np
from scipy.optimize import minimize

def func(x):
return (8 - x[0]) ** 2 - (7 - x[1]) ** 2 + 3 * x[1] ** 4

def grad(x):
return np.array([-2 * (8 - x[0]), -2 * (x[1] - 7) + 12 * x[1] ** 3])

def hess(x):
return np.array([[-2, 0], [0, -2 + 36 * x[1] ** 2]])

x0 = np.array([-3, 5])
res = minimize(func, x0, method='Newton-CG', jac=grad, hess=hess)

print(res)
```
